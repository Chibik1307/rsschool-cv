function points(arrayGames) {
	return arrayGames.reduce((acc, curr) => { 
	  const numbers = curr.split(":");
	  const [first, second] = numbers;

	  if (+first > +second) {
	    acc += 3;
	  }

	  if (+first === +second) {
	    acc += 1;
	  }
	  return acc;
	}, 0);
}
